def plot_plus_vertical_lines(x, y, *x_line_locations):
    y_limits = [y.min(), y.max()]
    args = []
    for x_loc in x_line_locations:
        args.extend([[x_loc, x_loc], y_limits, 'k:'])
    plt.plot(x, y, *args)
